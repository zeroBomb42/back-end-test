ล้าง docker image ให้ว่าง

docker stop $(docker ps -aq)
docker rm $(docker ps -aq)
docker rmi $(docker images -q)
docker volume prune -f
docker network prune -f
docker system prune -a -f --volumes

เช็คว่ามี container อะไรบ้างที่ run อยู่ใน docker 
docker ps

เช็คว่ามี container ทั้งหมดที่อยู่ใน docker 
docker ps -a

เอา backend และ backend_db ขึ้น docker
docker-compose up

เอา backend และ backend_db ลง docker
docker-compose down
